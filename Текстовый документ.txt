-- Добавляем поле photo в таблицу personal
ALTER TABLE personal 
ADD COLUMN photo LONGBLOB NULL AFTER telephone_number,
ADD COLUMN photo_path VARCHAR(255) NULL AFTER photo,
ADD COLUMN hire_date DATE NULL AFTER photo_path;

-- Добавляем поле is_active в таблицу users
ALTER TABLE users 
ADD COLUMN is_active BOOLEAN DEFAULT TRUE AFTER role_id;

        private void LoadAllStaffData()
        {
            try
            {
                using (var connection = DatabaseConnection.GetConnection())
                {
                    connection.Open();

                    string query = @"
                        SELECT 
                            p.id as PersonalId,
                            p.FIO,
                            p.job_title,
                            p.telephone_number,
                            p.email,
                            p.photo,
                            p.hire_date,
                            u.id as UserId,
                            u.login,
                            u.role_id,
                            u.is_active,
                            r.role_name
                        FROM personal p
                        LEFT JOIN users u ON p.id = u.personal_id
                        LEFT JOIN role r ON u.role_id = r.id
                        ORDER BY p.FIO";

                    using (var cmd = new MySqlCommand(query, connection))
                    {
                        MySqlDataAdapter adapter = new MySqlDataAdapter(cmd);
                        allStaffData = new DataTable();
                        adapter.Fill(allStaffData);
                    }
                }

                ApplyFilter();
                UpdateSummary();
                toolStripStatusLabel.Text = $"Загружено сотрудников: {allStaffData.Rows.Count}";
            }
            catch (Exception ex)
            {
                MessageBox.Show($"Ошибка загрузки данных: {ex.Message}", "Ошибка",
                    MessageBoxButtons.OK, MessageBoxIcon.Error);
            }
        }

        private void ApplyFilter()
        {
            if (allStaffData == null) return;

            var filtered = allStaffData.AsEnumerable();

            // Поиск по тексту
            string searchText = textBoxSearch.Text.Trim().ToLower();
            if (!string.IsNullOrEmpty(searchText))
            {
                filtered = filtered.Where(row =>
                    row.Field<string>("FIO")?.ToLower().Contains(searchText) == true ||
                    row.Field<string>("job_title")?.ToLower().Contains(searchText) == true ||
                    row.Field<string>("email")?.ToLower().Contains(searchText) == true ||
                    row.Field<string>("login")?.ToLower().Contains(searchText) == true);
            }

            // Фильтр по роли
            if (comboBoxRole.SelectedIndex > 0)
            {
                string selectedRole = comboBoxRole.SelectedItem.ToString();
                filtered = filtered.Where(row => 
                    row.Field<string>("role_name") == selectedRole);
            }

            // Фильтр по статусу
            if (comboBoxStatus.SelectedIndex == 1) // Активные
            {
                filtered = filtered.Where(row => 
                    row.Field<bool?>("is_active") == true);
            }
            else if (comboBoxStatus.SelectedIndex == 2) // Неактивные
            {
                filtered = filtered.Where(row => 
                    row.Field<bool?>("is_active") == false || 
                    row.Field<bool?>("is_active") == null);
            }

            DataTable filteredTable = filtered.Any() ? filtered.CopyToDataTable() : allStaffData.Clone();
            
            totalPages = (int)Math.Ceiling((double)filteredTable.Rows.Count / pageSize);
            if (totalPages == 0) totalPages = 1;
            if (currentPage > totalPages) currentPage = totalPages;
            if (currentPage < 1) currentPage = 1;

            LoadCurrentPage(filteredTable);
            UpdatePageInfo(filteredTable.Rows.Count);
        }

        private void LoadCurrentPage(DataTable filteredTable)
        {
            int startIndex = (currentPage - 1) * pageSize;
            int endIndex = Math.Min(startIndex + pageSize, filteredTable.Rows.Count);

            currentPageData = filteredTable.Clone();
            for (int i = startIndex; i < endIndex; i++)
            {
                currentPageData.ImportRow(filteredTable.Rows[i]);
            }

            dataGridViewStaff.Rows.Clear();

            foreach (DataRow row in currentPageData.Rows)
            {
                int rowIndex = dataGridViewStaff.Rows.Add();
                var dgvRow = dataGridViewStaff.Rows[rowIndex];

                // Фото
                if (row["photo"] != DBNull.Value)
                {
                    byte[] photoBytes = (byte[])row["photo"];
                    using (MemoryStream ms = new MemoryStream(photoBytes))
                    {
                        dgvRow.Cells["colPhoto"].Value = Image.FromStream(ms);
                    }
                }
                else
                {
                    // Заглушка для фото
                    dgvRow.Cells["colPhoto"].Value = CreateDefaultAvatar(row.Field<string>("FIO"));
                }

                dgvRow.Cells["colId"].Value = row["PersonalId"];
                dgvRow.Cells["colFIO"].Value = row["FIO"];
                dgvRow.Cells["colJobTitle"].Value = row["job_title"];
                dgvRow.Cells["colRole"].Value = row["role_name"] ?? "Нет доступа";
                dgvRow.Cells["colLogin"].Value = row["login"] ?? "-";
                dgvRow.Cells["colPhone"].Value = FormatPhoneNumber(row.Field<string>("telephone_number"));
                dgvRow.Cells["colEmail"].Value = row["email"] ?? "-";
                
                if (row["hire_date"] != DBNull.Value)
                {
                    DateTime hireDate = Convert.ToDateTime(row["hire_date"]);
                    dgvRow.Cells["colHireDate"].Value = hireDate.ToString("dd.MM.yyyy");
                }
                
                dgvRow.Cells["colIsActive"].Value = row.Field<bool?>("is_active") ?? false;
                dgvRow.Cells["colResetPassword"].Value = "🔄 Сбросить";
                
                dgvRow.Tag = row["UserId"] ?? DBNull.Value;
            }
        }

        private Image CreateDefaultAvatar(string fio)
        {
            Bitmap bmp = new Bitmap(50, 50);
            using (Graphics g = Graphics.FromImage(bmp))
            {
                g.Clear(Color.FromArgb(76, 145, 195));
                
                // Получаем первую букву ФИО
                string initials = "";
                if (!string.IsNullOrEmpty(fio))
                {
                    string[] parts = fio.Split(' ');
                    if (parts.Length > 0 && !string.IsNullOrEmpty(parts[0]))
                        initials += parts[0][0];
                    if (parts.Length > 1 && !string.IsNullOrEmpty(parts[1]))
                        initials += parts[1][0];
                }
                
                using (Font font = new Font("Segoe UI", 20, FontStyle.Bold))
                using (StringFormat sf = new StringFormat())
                {
                    sf.Alignment = StringAlignment.Center;
                    sf.LineAlignment = StringAlignment.Center;
                    
                    g.DrawString(initials, font, Brushes.White, 
                        new RectangleF(0, 0, 50, 50), sf);
                }
            }
            return bmp;
        }

        private string FormatPhoneNumber(string phone)
        {
            if (string.IsNullOrEmpty(phone)) return "-";
            
            string digits = new string(phone.Where(char.IsDigit).ToArray());
            
            if (digits.Length == 11)
            {
                return $"+7 ({digits.Substring(1, 3)}) {digits.Substring(4, 3)}-{digits.Substring(7, 2)}-{digits.Substring(9, 2)}";
            }
            
            return phone;
        }

        private void DeleteStaff(int staffId)
        {
            try
            {
                using (var connection = DatabaseConnection.GetConnection())
                {
                    connection.Open();
                    using (var transaction = connection.BeginTransaction())
                    {
                        try
                        {
                            // Удаляем пользователя
                            string deleteUser = "DELETE FROM users WHERE personal_id = @staffId";
                            using (var cmd = new MySqlCommand(deleteUser, connection, transaction))
                            {
                                cmd.Parameters.AddWithValue("@staffId", staffId);
                                cmd.ExecuteNonQuery();
                            }

                            // Удаляем сотрудника
                            string deleteStaff = "DELETE FROM personal WHERE id = @staffId";
                            using (var cmd = new MySqlCommand(deleteStaff, connection, transaction))
                            {
                                cmd.Parameters.AddWithValue("@staffId", staffId);
                                cmd.ExecuteNonQuery();
                            }

                            transaction.Commit();
                            MessageBox.Show("Сотрудник успешно удален!", "Успех",
                                MessageBoxButtons.OK, MessageBoxIcon.Information);
                            
                            LoadAllStaffData();
                        }
                        catch
                        {
                            transaction.Rollback();
                            throw;
                        }
                    }
                }
            }
            catch (MySqlException ex) when (ex.Number == 1451)
            {
                MessageBox.Show("Нельзя удалить сотрудника, который создавал заказы или бронирования.\n" +
                    "Сначала деактивируйте его учетную запись.",
                    "Ошибка удаления", MessageBoxButtons.OK, MessageBoxIcon.Error);
            }
            catch (Exception ex)
            {
                MessageBox.Show($"Ошибка при удалении: {ex.Message}", "Ошибка",
                    MessageBoxButtons.OK, MessageBoxIcon.Error);
            }
        }

        private void ResetPassword(int userId)
        {
            var result = MessageBox.Show(
                "Сбросить пароль пользователя?\n" +
                "Новый пароль будет установлен: 123456",
                "Подтверждение сброса пароля",
                MessageBoxButtons.YesNo,
                MessageBoxIcon.Question);

            if (result == DialogResult.Yes)
            {
                try
                {
                    using (var connection = DatabaseConnection.GetConnection())
                    {
                        connection.Open();
                        
                        string query = "UPDATE users SET password = @password WHERE id = @userId";
                        using (var cmd = new MySqlCommand(query, connection))
                        {
                            cmd.Parameters.AddWithValue("@password", GetPasswordHash("123456"));
                            cmd.Parameters.AddWithValue("@userId", userId);
                            
                            int rows = cmd.ExecuteNonQuery();
                            if (rows > 0)
                            {
                                MessageBox.Show("Пароль успешно сброшен на: 123456", "Успех",
                                    MessageBoxButtons.OK, MessageBoxIcon.Information);
                            }
                        }
                    }
                }
                catch (Exception ex)
                {
                    MessageBox.Show($"Ошибка при сбросе пароля: {ex.Message}", "Ошибка",
                        MessageBoxButtons.OK, MessageBoxIcon.Error);
                }
            }
        }

        private string GetPasswordHash(string password)
        {
            using (var sha256 = System.Security.Cryptography.SHA256.Create())
            {
                byte[] hash = sha256.ComputeHash(System.Text.Encoding.UTF8.GetBytes(password));
                return BitConverter.ToString(hash).Replace("-", "").ToLower();
            }
        }

                            r.role_name

                                {
                                    editingUserId = Convert.ToInt32(reader["UserId"]);
                                    checkBoxCreateAccount.Checked = true;
                                    textBoxLogin.Text = reader["login"].ToString();
                                    textBoxPassword.Text = "●●●●●●";

        private int GetRoleIndex(object roleId)
        {
            if (roleId == DBNull.Value) return -1;
            int id = Convert.ToInt32(roleId);
            switch (id)
            {
                case 1: return 0; // Администратор
                case 2: return 1; // Рецепшен
                case 3: return 2; // Менеджер

        private string FormatPhoneForMask(string phone)
        {
            if (string.IsNullOrEmpty(phone)) return "";
            
            string digits = new string(phone.Where(char.IsDigit).ToArray());
            if (digits.Length == 11 && digits.StartsWith("7"))
                digits = digits.Substring(1);
            
            return digits.Length == 10 ? digits : "";
        }

        private string GenerateLoginFromFIO(string fio)
        {
            // Транслитерация ФИО в латиницу
            string[] parts = fio.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length == 0) return "";
            
            string lastName = Transliterate(parts[0].ToLower());
            string firstName = parts.Length > 1 ? Transliterate(parts[1].ToLower()) : "";
            
            if (string.IsNullOrEmpty(firstName))
                return lastName;
            
            return $"{lastName}_{firstName[0]}";
        }

        private string Transliterate(string text)
        {
            Dictionary<char, string> translit = new Dictionary<char, string>
            {
                {'а', "a"}, {'б', "b"}, {'в', "v"}, {'г', "g"}, {'д', "d"},
                {'е', "e"}, {'ё', "e"}, {'ж', "zh"}, {'з', "z"}, {'и', "i"},
                {'й', "y"}, {'к', "k"}, {'л', "l"}, {'м', "m"}, {'н', "n"},
                {'о', "o"}, {'п', "p"}, {'р', "r"}, {'с', "s"}, {'т', "t"},
                {'у', "u"}, {'ф', "f"}, {'х', "h"}, {'ц', "ts"}, {'ч', "ch"},
                {'ш', "sh"}, {'щ', "sch"}, {'ъ', ""}, {'ы', "y"}, {'ь', ""},
                {'э', "e"}, {'ю', "yu"}, {'я', "ya"}
            };
            
            string result = "";
            foreach (char c in text)
            {
                if (translit.ContainsKey(c))
                    result += translit[c];
                else
                    result += c;
            }
            return result;
        }

        private Image ResizeImage(Image image, int width, int height)
        {
            var destRect = new Rectangle(0, 0, width, height);
            var destImage = new Bitmap(width, height);
            
            destImage.SetResolution(image.HorizontalResolution, image.VerticalResolution);
            
            using (var graphics = Graphics.FromImage(destImage))
            {
                graphics.CompositingMode = System.Drawing.Drawing2D.CompositingMode.SourceCopy;
                graphics.CompositingQuality = System.Drawing.Drawing2D.CompositingQuality.HighQuality;
                graphics.InterpolationMode = System.Drawing.Drawing2D.InterpolationMode.HighQualityBicubic;
                graphics.SmoothingMode = System.Drawing.Drawing2D.SmoothingMode.HighQuality;
                graphics.PixelOffsetMode = System.Drawing.Drawing2D.PixelOffsetMode.HighQuality;
                
                using (var wrapMode = new System.Drawing.Imaging.ImageAttributes())
                {
                    wrapMode.SetWrapMode(System.Drawing.Drawing2D.WrapMode.TileFlipXY);
                    graphics.DrawImage(image, destRect, 0, 0, image.Width, image.Height, GraphicsUnit.Pixel, wrapMode);
                }
            }
            
            return destImage;
        }

        private string GenerateRandomPassword()
        {
            Random rand = new Random();
            const string chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789";
            return new string(Enumerable.Repeat(chars, 8)
                .Select(s => s[rand.Next(s.Length)]).ToArray());

        private bool ValidateData()
        {
            // Проверка ФИО
            if (string.IsNullOrWhiteSpace(textBoxFIO.Text))
            {
                ShowError("Введите ФИО сотрудника", textBoxFIO);
                return false;
            }

            // Проверка должности
            if (string.IsNullOrWhiteSpace(textBoxJobTitle.Text))
            {
                ShowError("Введите должность сотрудника", textBoxJobTitle);
                return false;
            }

            // Проверка телефона
            string phone = GetCleanPhone();
            if (string.IsNullOrWhiteSpace(phone) || phone.Length != 10)
            {
                ShowError("Введите корректный номер телефона", maskedTextBoxPhone);
                return false;
            }

            // Проверка email
            if (!string.IsNullOrWhiteSpace(textBoxEmail.Text) && !IsValidEmail(textBoxEmail.Text))
            {
                ShowError("Введите корректный email адрес", textBoxEmail);
                return false;
            }

            // Проверка учетной записи
            if (checkBoxCreateAccount.Checked)
            {
                if (string.IsNullOrWhiteSpace(textBoxLogin.Text))
                {
                    ShowError("Введите логин для учетной записи", textBoxLogin);
                    return false;
                }

                if (textBoxLogin.Text.Length < 3)
                {
                    ShowError("Логин должен содержать минимум 3 символа", textBoxLogin);
                    return false;
                }

                if (currentMode == FormMode.Add && string.IsNullOrWhiteSpace(textBoxPassword.Text))
                {
                    ShowError("Введите пароль для учетной записи", textBoxPassword);
                    return false;
                }

                if (comboBoxRole.SelectedIndex == -1)
                {
                    ShowError("Выберите роль пользователя", comboBoxRole);
                    return false;
                }
            }

            return true;
        }

        private void AddStaff()
        {
            try
            {
                using (var connection = DatabaseConnection.GetConnection())
                {
                    connection.Open();
                    using (var transaction = connection.BeginTransaction())
                    {
                        try
                        {
                            // 1. Добавляем сотрудника
                            string insertPersonal = @"
                                INSERT INTO personal (
                                    FIO, job_title, telephone_number, email, 
                                    hire_date, photo
                                ) VALUES (
                                    @fio, @jobTitle, @phone, @email, 
                                    @hireDate, @photo
                                );
                                SELECT LAST_INSERT_ID();";

                            int personalId;
                            using (var cmd = new MySqlCommand(insertPersonal, connection, transaction))
                            {
                                cmd.Parameters.AddWithValue("@fio", textBoxFIO.Text.Trim());
                                cmd.Parameters.AddWithValue("@jobTitle", textBoxJobTitle.Text.Trim());
                                cmd.Parameters.AddWithValue("@phone", FormatPhoneForDB(GetCleanPhone()));
                                cmd.Parameters.AddWithValue("@email", 
                                    string.IsNullOrWhiteSpace(textBoxEmail.Text) ? 
                                    DBNull.Value : (object)textBoxEmail.Text.Trim());
                                cmd.Parameters.AddWithValue("@hireDate", dateTimePickerHireDate.Value.Date);
                                cmd.Parameters.AddWithValue("@photo", 
                                    currentPhotoBytes ?? (object)DBNull.Value);
                                
                                personalId = Convert.ToInt32(cmd.ExecuteScalar());
                            }

                            // 2. Создаем учетную запись если нужно
                            if (checkBoxCreateAccount.Checked)
                            {
                                string insertUser = @"
                                    INSERT INTO users (
                                        personal_id, login, password, role_id, is_active
                                    ) VALUES (
                                        @personalId, @login, @password, @roleId, @isActive
                                    )";

                                using (var cmd = new MySqlCommand(insertUser, connection, transaction))
                                {
                                    cmd.Parameters.AddWithValue("@personalId", personalId);
                                    cmd.Parameters.AddWithValue("@login", textBoxLogin.Text.Trim());
                                    cmd.Parameters.AddWithValue("@password", GetPasswordHash(textBoxPassword.Text));
                                    cmd.Parameters.AddWithValue("@roleId", comboBoxRole.SelectedIndex + 1);
                                    cmd.Parameters.AddWithValue("@isActive", checkBoxIsActive.Checked);
                                    
                                    cmd.ExecuteNonQuery();
                                }
                            }

                            transaction.Commit();
                            
                            DataChanged = true;
                            MessageBox.Show("Сотрудник успешно добавлен!", "Успех",
                                MessageBoxButtons.OK, MessageBoxIcon.Information);
                            
                            this.DialogResult = DialogResult.OK;
                            this.Close();
                        }
                        catch
                        {
                            transaction.Rollback();
                            throw;
                        }
                    }
                }
            }
            catch (MySqlException ex) when (ex.Number == 1062)
            {
                if (ex.Message.Contains("login"))
                {
                    MessageBox.Show("Пользователь с таким логином уже существует.", 
                        "Ошибка", MessageBoxButtons.OK, MessageBoxIcon.Error);
                }
                else if (ex.Message.Contains("email"))
                {
                    MessageBox.Show("Сотрудник с таким email уже существует.", 
                        "Ошибка", MessageBoxButtons.OK, MessageBoxIcon.Error);
                }
                else
                {
                    MessageBox.Show($"Ошибка при добавлении: {ex.Message}", 
                        "Ошибка", MessageBoxButtons.OK, MessageBoxIcon.Error);
                }
            }
            catch (Exception ex)
            {
                MessageBox.Show($"Ошибка при добавлении: {ex.Message}", 
                    "Ошибка", MessageBoxButtons.OK, MessageBoxIcon.Error);
            }
        }

        private string FormatPhoneForDB(string cleanPhone)
        {
            if (cleanPhone.Length == 10)
            {
                return $"+7({cleanPhone.Substring(0, 3)}){cleanPhone.Substring(3, 3)}-{cleanPhone.Substring(6, 2)}-{cleanPhone.Substring(8)}";
            }
            return cleanPhone;
        }

        private string GetPasswordHash(string password)
        {
            using (var sha256 = System.Security.Cryptography.SHA256.Create())
            {
                byte[] hash = sha256.ComputeHash(System.Text.Encoding.UTF8.GetBytes(password));
                return BitConverter.ToString(hash).Replace("-", "").ToLower();
            }
        }
